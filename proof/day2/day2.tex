\[
    \begin{split}
       A = \{1,2\} \\ B = \{2,3\} \\
       (A \times B) \cap (B \times B) = \\ 
       \{\{1,2\},\{1,3\},\{2,2\},\{2,3\}\} \cap \{\{2,2\},\{2,3\},\{3,2\},\{3,3\}\}        
        \\= \{\{2,2\},\{2,3\}\}
    \end{split}
\]
